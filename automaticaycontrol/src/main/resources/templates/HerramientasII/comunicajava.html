           private static boolean PasswordCorrect(char[] input) {
                boolean correcto = true;
                char[] clave = {'R', 'u', 'b', 'e', 'n', '1', '2', '3'};
                if (input.length != clave.length) {
        //            System.out.println("" + clave.length);
        //            System.out.println("" + input);
                    correcto = false;
                } else {
                    for (int i = 0; i < input.length; i++) {
                        if (input[i] != clave[i]) {
                            correcto = false;
                        }
                    }
                }

                for (int i = 0; i < clave.length; i++) {
                    clave[i] = 0;
                }

                return correcto;
            }
