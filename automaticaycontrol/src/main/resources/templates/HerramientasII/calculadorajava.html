public class CalculadoraBasica extends javax.swing.JFrame {

  public CalculadoraBasica() {
        initComponents();
        setLocationRelativeTo(null);
        Calendar c = Calendar.getInstance();
        int anio = c.get(Calendar.YEAR);
        int mes = c.get(Calendar.MONTH);
        int dia = c.get(Calendar.DAY_OF_MONTH);
        int hora = c.get(Calendar.HOUR);
        int min = c.get(Calendar.MINUTE);
        int seg = c.get(Calendar.SECOND);
        tiempo.setText(String.valueOf(anio) + "/"
           + String.valueOf(mes+1) + "/"
           + String.valueOf(dia) + "     " + String.valueOf(hora)
           + ":" + String.valueOf(min) + ":" + String.valueOf(seg));
        titulo.setText("Ruben@ngel Designer");
    }

  private void SumarActionPerformed(java.awt.event.ActionEvent evt) {
        double varX = Double.parseDouble(var_X.getText());
        double varY = Double.parseDouble(var_Y.getText());
//        double resultado = varX*varY;
//        result.setText(String.format("%.2f", resultado));
        result.setText(String.format("%.2f", varX+varY));
    }

private void RestarActionPerformed(java.awt.event.ActionEvent evt) {
        double varX = Double.parseDouble(var_X.getText());
        //double varY = Double.parseDouble(var_Y.getText());
//        double resultado = varX*varY;
//        result.setText(String.format("%.2f", resultado));
        result.setText(String.format("%.2f", Math.sin(varX*Math.PI/180)));
    }
