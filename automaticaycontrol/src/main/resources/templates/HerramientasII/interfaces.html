namespace HerenciaAnimales
{

    interface IMamiferosTerrestres
    {
        int NumeroPatas();
    }

    interface IAnimalesyDeportes
    {
        string TipoDeporte();

        string EsOlimpico();
    }

    interface ISaltaConPatas
    {
        int CuantasPatas();
    }

    public class Mamiferos
    {
        private string nombre_mamifero;

        public Mamiferos(string nombre)
        {
            nombre_mamifero = nombre;
        }

        public void getNombre() {
            Console.WriteLine("El nombre del mamífero es: " + nombre_mamifero);
        }



        public void respirar()
        {
            Console.WriteLine("Puedo respirar");
        }

        public void CuidarCrias()
        {
            Console.WriteLine("Cuido me mis crias hasta que se valagan solas");
        }

        public virtual void pensar()
        {
            Console.WriteLine("Pensamiento básico instintivo");
        }
    }
}


namespace HerenciaAnimales
{
    public class Humanos: Mamiferos
    {
        public Humanos(string nombre) : base(nombre)
        {
        }

        public override void pensar()
        {
            Console.WriteLine("Soy capaz de pensar");
        }
    }
}


namespace HerenciaAnimales
{
    public class Caballo:Mamiferos, IMamiferosTerrestres, IAnimalesyDeportes, ISaltaConPatas
    {
        public Caballo(string nombre) : base(nombre)
        {
        }

        public void galopar()
        {
            Console.WriteLine("Me muevo al galope");
        }

        public int NumeroPatas()
        {
            return 4;
        }

        public string TipoDeporte()
        {
            return "Equitación";
        }

        public string EsOlimpico()
        {
            return "es olímpico";
        }

        public int CuantasPatas()
        {
            return 2;
        }

        public override void pensar()
        {
            Console.WriteLine("Pensamiento básico");
        }
    }
}
